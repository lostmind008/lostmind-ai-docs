```bash
git clone https://github.com/example/github-repository-analyzer.git
```

### 2. Environment Variables Configuration

**Option A: .env File (Recommended for Development)**
```bash
# Create .env file
cat > .env << EOF
# GitHub API Configuration
GITHUB_TOKEN=ghp_your_github_personal_access_token_here

# Gemini AI Configuration  
GOOGLE_API_KEY=your_gemini_api_key_from_ai_studio

# Optional: Application Configuration
LOG_LEVEL=INFO
OUTPUT_DIR=analysis_output
MAX_CONCURRENT_ANALYSIS=10
MEMORY_LIMIT_MB=1000
ANALYSIS_TIMEOUT=300
EOF

# Secure the .env file
chmod 600 .env
```

**Option B: System Environment Variables (Recommended for Production)**
```bash
# Linux/macOS - Add to ~/.bashrc or ~/.zshrc
export GITHUB_TOKEN="ghp_your_token_here"
export GOOGLE_API_KEY="your_api_key_here"
export LOG_LEVEL="INFO"

# Windows - Use System Properties or PowerShell
[Environment]::SetEnvironmentVariable("GITHUB_TOKEN", "ghp_your_token", "User")
[Environment]::SetEnvironmentVariable("GOOGLE_API_KEY", "your_api_key", "User")
```

## ðŸ³ Docker Deployment

### Standard Docker Deployment

**1. Create Dockerfile** (already included in project):
```dockerfile
FROM python:3.11-slim

# Set working directory
WORKDIR /app

# Install system dependencies
RUN apt-get update && apt-get install -y \
    git \
    curl \
    && rm -rf /var/lib/apt/lists/*

# Copy requirements and install Python dependencies
COPY requirements.txt .
RUN pip install --no-cache-dir --upgrade pip \
    && pip install --no-cache-dir -r requirements.txt

# Copy application code
COPY . .

# Install application
RUN pip install -e .

# Create non-root user for security
RUN useradd --create-home --shell /bin/bash analyzer
RUN chown -R analyzer:analyzer /app
USER analyzer

# Set environment variables
ENV PYTHONPATH=/app/src
ENV PYTHONUNBUFFERED=1

# Expose port (if needed for web interface)
EXPOSE 8080

# Default command
CMD ["github-analyzer", "--concurrent", "--max-concurrent", "10"]
```

**2. Build and Run Docker Image**:
```bash
# Build image
docker build -t github-analyzer:latest .

# Run with environment variables
docker run -d \
  --name github-analyzer \
  -e GITHUB_TOKEN="$GITHUB_TOKEN" \
  -e GOOGLE_API_KEY="$GOOGLE_API_KEY" \
  -v $(pwd)/analysis_output:/app/analysis_output \
  github-analyzer:latest

# Check logs
docker logs github-analyzer

# Interactive run for testing
docker run -it --rm \
  -e GITHUB_TOKEN="$GITHUB_TOKEN" \
  -e GOOGLE_API_KEY="$GOOGLE_API_KEY" \
  -v $(pwd)/analysis_output:/app/analysis_output \
  github-analyzer:latest \
  github-analyzer -u octocat --concurrent --log-level DEBUG
```
