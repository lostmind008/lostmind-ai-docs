3. **Show Conversation Patterns**:
   ```bash
   echo "üßµ **Conversation Patterns:**"
   echo "----------------------------"

   if [ -f "$PROJECT_CONTEXT/conversation_threads.json" ]; then
       python3 -c "
import json
from collections import Counter

try:
    with open('$PROJECT_CONTEXT/conversation_threads.json', 'r') as f:
        data = json.load(f)

    if 'threads' in data and isinstance(data['threads'], list):
        threads = data['threads']
        print(f'üìä Total Conversation Threads: {len(threads)}')

        # Analyze conversation topics
        topics = []
        for thread in threads[:20]:  # Analyze recent threads
            if isinstance(thread, dict) and 'content' in thread:
                content = str(thread.get('content', '')).lower()
                # Simple topic detection
                if 'bug' in content or 'error' in content or 'fix' in content:
                    topics.append('bug_fixing')
                elif 'feature' in content or 'implement' in content or 'add' in content:
                    topics.append('feature_development')
                elif 'test' in content or 'testing' in content:
                    topics.append('testing')
                elif 'deploy' in content or 'production' in content:
                    topics.append('deployment')
                elif 'refactor' in content or 'improve' in content:
                    topics.append('refactoring')
                else:
                    topics.append('general_discussion')

        if topics:
            topic_counts = Counter(topics)
            print('')
            print('üìà **Topic Distribution:**')
            for topic, count in topic_counts.most_common(5):
                percentage = (count / len(topics)) * 100
                topic_name = topic.replace('_', ' ').title()
                print(f'   {topic_name}: {count} threads ({percentage:.1f}%)')

        print('')

    else:
        print('No conversation patterns available yet')

except Exception as e:
    print(f'Could not analyze conversation patterns: {e}')
"
   else
       echo "No conversation thread data available"
   fi
   ```
