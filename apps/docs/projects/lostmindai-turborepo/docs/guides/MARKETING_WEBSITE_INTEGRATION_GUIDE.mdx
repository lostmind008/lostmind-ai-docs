   import { Analytics } from '@vercel/analytics/react';
   import posthog from 'posthog-js';

   export class MarketingAnalytics {
     private static instance: MarketingAnalytics;
     
     static getInstance() {
       if (!this.instance) {
         this.instance = new MarketingAnalytics();
       }
       return this.instance;
     }

     initialize() {
       // PostHog initialization
       if (typeof window !== 'undefined') {
         posthog.init(process.env.NEXT_PUBLIC_POSTHOG_KEY!, {
           api_host: process.env.NEXT_PUBLIC_POSTHOG_HOST,
           loaded: (posthog) => {
             if (process.env.NODE_ENV === 'development') posthog.debug();
           }
         });
       }
     }

     trackConversion(event: string, properties?: Record<string, any>) {
       posthog.capture(event, properties);
       
       // Google Analytics 4
       if (typeof gtag !== 'undefined') {
         gtag('event', event, properties);
       }
     }

     trackPageView(url: string) {
       posthog.capture('$pageview', { url });
     }
   }

### Phase 4: Content Management & Blog

1. **MDX Blog Setup**
   ```typescript
   // apps/marketing/src/lib/content.ts
   import fs from 'fs';
   import path from 'path';
   import matter from 'gray-matter';

   export interface BlogPost {
     slug: string;
     title: string;
     description: string;
     publishedAt: string;
     author: string;
     tags: string[];
     content: string;
   }

   export async function getBlogPosts(): Promise<BlogPost[]> {
     const postsDirectory = path.join(process.cwd(), 'content/blog');
     const filenames = fs.readdirSync(postsDirectory);

     const posts = filenames
       .filter(name => name.endsWith('.mdx'))
       .map((name) => {
         const fullPath = path.join(postsDirectory, name);
         const fileContents = fs.readFileSync(fullPath, 'utf8');
         const { data, content } = matter(fileContents);

         return {
           slug: name.replace(/\.mdx$/, ''),
           title: data.title,
           description: data.description,
           publishedAt: data.publishedAt,
           author: data.author,
           tags: data.tags || [],
           content
         };
       })
       .sort((a, b) => 
         new Date(b.publishedAt).getTime() - new Date(a.publishedAt).getTime()
       );

     return posts;
   }
   ```

// apps/marketing/src/lib/ab-testing.ts
import posthog from 'posthog-js';

export function useABTest(testName: string, variants: string[]) {
  const [variant, setVariant] = React.useState<string>('control');

  React.useEffect(() => {
    if (typeof window !== 'undefined') {
      const testVariant = posthog.getFeatureFlag(testName) as string;
      if (variants.includes(testVariant)) {
        setVariant(testVariant);
      }
    }
  }, [testName, variants]);

  return variant;
}
