## 🛠️ API Reference

### Programmatic Usage

```python
from github_analyzer import GitHubAnalyzer, ConcurrentGitHubAnalyzer
from github_analyzer.utils.config import get_config

# Sequential analysis
analyzer = GitHubAnalyzer(get_config())
analyzer.analyze_user_repositories('username')

# High-performance concurrent analysis
import asyncio
from github_analyzer.core.concurrent_analyzer import ConcurrencyConfig

config = get_config()
concurrency_config = ConcurrencyConfig(
    MAX_CONCURRENT_ANALYSIS=15,
    MEMORY_LIMIT_MB=1000
)

async def analyze_concurrent():
    async with ConcurrentGitHubAnalyzer(config, concurrency_config) as analyzer:
        await analyzer.analyze_user_repositories_async('username')
        performance = analyzer.get_performance_summary()
        print(f"Analyzed {performance['total_repositories']} repos in {performance['analysis_time_seconds']:.1f}s")

asyncio.run(analyze_concurrent())
```
