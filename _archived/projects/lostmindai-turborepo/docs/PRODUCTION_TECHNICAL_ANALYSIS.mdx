```python
# services/ai-compute/main.py
from fastapi import FastAPI, Depends
from functools import lru_cache
from typing import Annotated

# Dependency injection pattern
class ServiceContainer:
    def __init__(self):
        self._ai_service: Optional[AIService] = None
        self._rag_service: Optional[RAGService] = None
    
    async def get_ai_service(self) -> AIService:
        if self._ai_service is None:
            self._ai_service = AIService(api_key=settings.GEMINI_API_KEY)
        return self._ai_service
    
    async def get_rag_service(self) -> RAGService:
        if self._rag_service is None:
            ai_service = await self.get_ai_service()
            self._rag_service = RAGService(
                database_url=settings.DATABASE_URL,
                ai_service=ai_service
            )
            await self._rag_service.initialize()
        return self._rag_service

# Application container
container = ServiceContainer()

# Dependency functions
async def get_ai_service() -> AIService:
    return await container.get_ai_service()

async def get_rag_service() -> RAGService:
    return await container.get_rag_service()

# Type aliases for cleaner endpoint signatures
AIServiceDep = Annotated[AIService, Depends(get_ai_service)]
RAGServiceDep = Annotated[RAGService, Depends(get_rag_service)]

# Updated endpoints
@app.post("/generate/text")
async def generate_text(
    request: TextGenerationRequest,
    ai_service: AIServiceDep
):
    try:
        response = await ai_service.generate_text(
            prompt=request.prompt,
            temperature=request.temperature,
            max_tokens=request.max_tokens
        )
        return response
    except Exception as e:
        logger.error("Text generation failed", error=str(e))
        raise HTTPException(status_code=500, detail=str(e))
```
